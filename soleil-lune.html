    function jd(date) { return date.getTime() / 86400000 + 2440587.5; }
    function sunPos(date) {
    const n = jd(date) - 2451545.0;
    const L = ((280.460 + 0.9856474 * n) % 360 + 360) % 360;
    const g = ((357.528 + 0.9856003 * n) % 360 + 360) % 360;
    const lam = (L + 1.915 * Math.sin(g*D2R) + 0.020 * Math.sin(2*g*D2R)) * D2R;
    const eps = (23.439 - 0.0000004 * n) * D2R;
    const dec = Math.asin(Math.sin(eps) * Math.sin(lam)) * R2D;
    const RA = Math.atan2(Math.cos(eps)*Math.sin(lam), Math.cos(lam)) * R2D;
    const GMST = ((280.46061837 + 360.98564736629 * n) % 360 + 360) % 360;
    let lon = ((RA - GMST) % 360 + 360) % 360;
    if (lon > 180) lon -= 360;
    return { lat: dec, lon };
    }

    const ms = ((357.528 + 0.9856003*n) % 360 + 360) % 360;
    const lDeg = L0 + 6.289*Math.sin(M*D2R) - 1.274*Math.sin((2*D-M)*D2R)
    + 0.658*Math.sin(2*D*D2R) - 0.186*Math.sin(ms*D2R)
    - 0.059*Math.sin((2*M-2*D)*D2R) - 0.057*Math.sin((M-2*D+ms)*D2R)
    + 0.053*Math.sin((M+2*D)*D2R) + 0.046*Math.sin((2*D-ms)*D2R)
    + 0.041*Math.sin((M-ms)*D2R) - 0.035*Math.sin(F*D2R)
    - 0.031*Math.sin((M+ms)*D2R) + 0.017*Math.sin(2*F*D2R);
    const beta = 5.128*Math.sin(F*D2R) + 0.2806*Math.sin((M+F)*D2R)
    + 0.2777*Math.sin((M-F)*D2R) + 0.1732*Math.sin((2*D-F)*D2R);
    const eps = 23.439 * D2R;
    const l = lDeg * D2R, b = beta * D2R;
    const sinDec = Math.sin(eps)*Math.sin(l)*Math.cos(b) + Math.cos(eps)*Math.sin(b);
    const dec = Math.asin(Math.max(-1,Math.min(1,sinDec))) * R2D;
    const RA = Math.atan2(Math.cos(eps)*Math.sin(l)*Math.cos(b)-Math.sin(eps)*Math.sin(b), Math.cos(l)*Math.cos(b)) * R2D;
    const GMST = ((280.46061837 + 360.98564736629*n) % 360+360) % 360;
    let lon = ((RA - GMST) % 360 + 360) % 360;
    if (lon > 180) lon -= 360;
    return { lat: dec, lon };
    }
    function moonPhase(date) {
    // Calcul basé sur l'élongation réelle Lune-Soleil en écliptique
    const n = jd(date) - 2451545.0;
    // Longitude écliptique du Soleil
    const Ls = ((280.460 + 0.9856474*n) % 360 + 360) % 360;
    const gs = ((357.528 + 0.9856003*n) % 360 + 360) % 360;
    const lonSun = (Ls + 1.915*Math.sin(gs*D2R) + 0.020*Math.sin(2*gs*D2R) + 360) % 360;
    // Longitude écliptique de la Lune
    const L0m = ((218.316 + 13.176396*n) % 360 + 360) % 360;
    const Mm = ((134.963 + 13.064993*n) % 360 + 360) % 360;
    const Fm = ((93.272 + 13.229350*n) % 360 + 360) % 360;
    const Dm = ((297.850 + 12.190749*n) % 360 + 360) % 360;
    const lonMoon = (L0m
    + 6.289*Math.sin(Mm*D2R)
    - 1.274*Math.sin((2*Dm-Mm)*D2R)
    + 0.658*Math.sin(2*Dm*D2R)
    - 0.214*Math.sin(2*Fm*D2R)
    - 0.110*Math.sin(Dm*D2R) + 360) % 360;
    // Élongation = angle Lune par rapport au Soleil [0–360°)
    const elong = ((lonMoon - lonSun) % 360 + 360) % 360;
    // Illumination
    const illum = (1 - Math.cos(elong * D2R)) / 2;
    // Âge approximatif en jours
    const age = elong / 360 * 29.53058867;
    // Nom
    let name;
    if (elong < 11) name = 'Nouvelle Lune \u{1F311}';
    else if (elong < 84) name = 'Croissant croissant \u{1F312}';
    else if (elong < 96) name = 'Premier Quartier \u{1F313}';
    else if (elong < 169) name = 'Gibbeuse croissante \u{1F314}';
    else if (elong < 191) name = 'Pleine Lune \u{1F315}';
    else if (elong < 264) name = 'Gibbeuse décroissante \u{1F316}';
    else if (elong < 276) name = 'Dernier Quartier \u{1F317}';
    else if (elong < 349) name = 'Croissant décroissant \u{1F318}';
    else name = 'Nouvelle Lune \u{1F311}';
    return { age, illum, name, elong };
    }

    function planetPos(date) {
    const JD = jd(date);
    const T = (JD - 2451545.0) / 36525.0;
    const n = JD - 2451545.0;
    const eps = 23.439 - 0.0000004 * n;
    // Earth heliocentric (for geocentric conversion)
    const Me = ((357.529 + 0.98560028*n) % 360 + 360) % 360;
    const Le = ((280.459 + 0.98564736*n) % 360 + 360) % 360;
    const Ce = 1.9146*Math.sin(Me*D2R) + 0.01993*Math.sin(2*Me*D2R);
    const Re = 1.00014 - 0.01671*Math.cos(Me*D2R) - 0.00014*Math.cos(2*Me*D2R);
    const lonE = (Le + Ce) * D2R;
    const Xe = Re * Math.cos(lonE);
    const Ye = Re * Math.sin(lonE);
    function eclToGeo(lonDeg, latDeg) {
    const l = lonDeg * D2R, b = latDeg * D2R, e = eps * D2R;
    const sinDec = Math.sin(b)*Math.cos(e) + Math.cos(b)*Math.sin(e)*Math.sin(l);
    const dec = Math.asin(Math.max(-1,Math.min(1,sinDec))) * R2D;
    const RA = Math.atan2(Math.sin(l)*Math.cos(e) - Math.tan(b)*Math.sin(e), Math.cos(l)) * R2D;
    const GMST = ((280.46061837 + 360.98564736629*n) % 360 + 360) % 360;
    let lon = ((RA - GMST) % 360 + 360) % 360;
    if (lon > 180) lon -= 360;
    return { lat: dec, lon };
    }
    function planet(a0, e0, i0, Om0, w0, M0, dM) {
    const M = ((M0 + dM * n) % 360 + 360) % 360;
    const Om = Om0 * D2R;
    const i = i0 * D2R;
    const w = w0 * D2R;
    // Kepler
    let E = M * D2R;
    for (let k=0; k<8; k++) E = M*D2R + e0*Math.sin(E);
    const nu = 2*Math.atan2(Math.sqrt(1+e0)*Math.sin(E/2), Math.sqrt(1-e0)*Math.cos(E/2));
    const r = a0*(1 - e0*Math.cos(E));
    const xh = r*(Math.cos(Om)*Math.cos(nu+w) - Math.sin(Om)*Math.sin(nu+w)*Math.cos(i));
    const yh = r*(Math.sin(Om)*Math.cos(nu+w) + Math.cos(Om)*Math.sin(nu+w)*Math.cos(i));
    const zh = r*(Math.sin(nu+w)*Math.sin(i));
    // Geocentric
    const xg = xh - Xe, yg = yh - Ye, zg = zh;
    const lonGeo = (Math.atan2(yg, xg)*R2D + 360) % 360;
    const latGeo = Math.atan2(zg, Math.sqrt(xg*xg+yg*yg)) * R2D;
    return eclToGeo(lonGeo, latGeo);
    }

    function terminatorPoly(sLat, sLon, altDeg) {
    const zenith = 90 - altDeg; // altDeg<=0 → zenith>=90
    const cosZ = Math.cos(zenith * D2R);
    const sinZ = Math.sin(zenith * D2R);
    const sinS = Math.sin(sLat * D2R);
    const cosS = Math.cos(sLat * D2R);
    const pts = [];
    for (let az = 0; az < 360; az += 1) {
    const a = az * D2R;
    const sinLat = sinS*cosZ + cosS*sinZ*Math.cos(a);
    const lat = Math.asin(Math.max(-1, Math.min(1, sinLat))) * R2D;
    const dLon = Math.atan2(sinZ*Math.sin(a)*cosS, cosZ - sinS*Math.sin(lat*D2R)) * R2D;
    let lon = sLon + dLon;
    while (lon > 180) lon -= 360;
    while (lon < -180) lon += 360;
    pts.push([lat, lon]);
    }
    pts.sort((a, b) => a[1] - b[1]);
    const pole = sLat >= 0 ? -90 : 90;
    return [...pts, [pole, 180], [pole, -180]];
    }

    function drawMoonCanvas(elong) {
    // elong: élongation en degrés [0-360)
    // 0=nouvelle, 90=premier quartier, 180=pleine, 270=dernier quartier
    // Côté droit éclairé 0→180 (croissant), côté gauche éclairé 180→360 (décroissant)
    const c = document.getElementById('moonCanvas');
    const cx = c.getContext('2d');
    const W=136, R=60, ox=68, oy=68;
    cx.clearRect(0,0,W,W);
    // Fond espace
    cx.fillStyle='#030810';
    cx.beginPath(); cx.arc(ox,oy,R,0,2*Math.PI); cx.fill();
    // Étoiles de fond
    const stars=[[12,18,0.6],[30,10,0.5],[52,22,0.7],[78,8,0.5],[94,18,0.6],[110,12,0.5],
    [122,28,0.7],[10,50,0.5],[126,52,0.6],[14,80,0.6],[60,90,0.7],[106,82,0.5],
    [128,90,0.6],[38,112,0.5],[82,122,0.6],[118,116,0.5],[24,130,0.6],[92,134,0.5]];
    stars.forEach(([sx,sy,r])=>{
    cx.globalAlpha=0.22+r*0.28;
    cx.fillStyle='rgba(200,220,255,0.9)';
    cx.beginPath(); cx.arc(sx,sy,r,0,2*Math.PI); cx.fill();
    });
    cx.globalAlpha=1;
    // Clip sur le disque
    cx.save();
    cx.beginPath(); cx.arc(ox,oy,R,0,2*Math.PI); cx.clip();
    // Surface lunaire
    const g=cx.createRadialGradient(ox-16,oy-16,4,ox,oy,R);
    g.addColorStop(0,'#e8f0f8'); g.addColorStop(0.5,'#c0d0e4'); g.addColorStop(1,'#98b0c8');
    cx.fillStyle=g; cx.fillRect(0,0,W,W);
    // Mers lunaires
    const mare=[
    [72,50,16,11,0.22],[86,62,11,8,0.20],[84,78,10,7,0.22],
    [62,66,9,6,0.18],[96,70,8,5,0.16],[68,88,13,8,0.20],[82,92,8,5,0.18]
    ];
    mare.forEach(([mx,my,rx,ry,op])=>{
    cx.fillStyle=`rgba(90,110,150,${op})`;
    cx.beginPath(); cx.ellipse(mx,my,rx,ry,0,0,2*Math.PI); cx.fill();
    });
    // ---- Ombre correcte basée sur l'élongation ----
    // Le terminateur est une ellipse verticale dont l'axe X = R * |cos(elong)|
    // Phase croissante (0-180) : côté droit éclairé → ombre sur la moitié gauche
    // Phase décroissante (180-360) : côté gauche éclairé → ombre sur la moitié droite
    const cosE = Math.cos(elong * D2R); // +1 à 0° (neuve), 0 à 90°, -1 à 180° (pleine), 0 à 270°, +1 à 360°
    cx.fillStyle = '#020710';
    if (elong < 180) {
    // Phase croissante : moitié gauche sombre + terminateur
    cx.beginPath();
    cx.arc(ox, oy, R, Math.PI/2, 3*Math.PI/2, false); // demi-cercle gauche
    if (cosE >= 0) {
    // 0→90° : cosE positif, terminateur ellipse bombée à gauche (ombre grande)
    cx.ellipse(ox, oy, cosE*R, R, 0, 3*Math.PI/2, Math.PI/2, false);
    } else {
    // 90→180° : cosE négatif, terminateur ellipse bombée à droite (ombre petite)
    cx.ellipse(ox, oy, -cosE*R, R, 0, Math.PI/2, 3*Math.PI/2, true);
    }
    cx.closePath(); cx.fill();
    } else {
    // Phase décroissante : moitié droite sombre + terminateur
    cx.beginPath();
    cx.arc(ox, oy, R, 3*Math.PI/2, Math.PI/2, false); // demi-cercle droit
    if (cosE <= 0) {
    // 180→270° : |cosE| positif, terminateur bombé à droite (ombre grande)
    cx.ellipse(ox, oy, -cosE*R, R, 0, Math.PI/2, 3*Math.PI/2, false);
    } else {
    // 270→360° : cosE positif, terminateur bombé à gauche (ombre petite)
    cx.ellipse(ox, oy, cosE*R, R, 0, 3*Math.PI/2, Math.PI/2, true);
    }
    cx.closePath(); cx.fill();
    }
    cx.restore();
    // Halo atmosphérique
    const gr=cx.createRadialGradient(ox,oy,R-1,ox,oy,R+12);
    gr.addColorStop(0,'rgba(160,200,255,0.15)'); gr.addColorStop(1,'rgba(0,0,0,0)');
    cx.fillStyle=gr; cx.beginPath(); cx.arc(ox,oy,R+12,0,2*Math.PI); cx.fill();
    // Contour
    cx.strokeStyle='rgba(90,140,210,0.3)'; cx.lineWidth=1;
    cx.beginPath(); cx.arc(ox,oy,R,0,2*Math.PI); cx.stroke();
    }

    }
